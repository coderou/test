    computed:{
      ...mapState({
        categoryList:(state)=>state.home.categoryList,
      })
    },
    method:{
      ...mapAction(['getBaseCategoryList'])
    },

    }

5.vuex的module基本用法:
  #store/index.js
  {
    import Vue from 'vue';
    import Vuex from 'vuex';
    import modules from './modules.js';
    Vue.use(Vuex)

    export default Vuex.Store({
      modules
    })
  }
  #store/modules/index.js
  {
    import home from './home.js';
    export default {
      home,
    }
  }
  #store/modules/home.js
  {
    import {reqGetBaseCategoryList} from '@/api';
    const state={
      categoryList:[];
    };
    const actions={
      etBaseCategoryList({commit}){
        reqGetBaseCategoryList().then((data)=>{
          commit('SETLIST',data)
        })
      }
    };
    const mutations={
      SETLIST(state,data){
        state.categoryList=data
      }
    };
    const getters={};
    export default {
      state,
      actions,
      mutations,
      getters,
    }

  }
